Lemma nth_error_ok :
  forall A i (l:list A),
    i < length l ->
    nth_error l i <> None.

Lemma list_nth_some :
  forall A (l:list A) i,
    i < length l ->
    exists x, nth_error l i = Some x.

Lemma list_nth_none:
  forall A (l:list A) i,
    length l <= i ->
    nth_error l i = None.

Lemma option_exist :
  forall A (o:option A),
    o <> None ->
    exists a, o = Some a.

Lemma list_nth_eq :
  forall A (la:list A) lb,
    ( forall i, nth_error la i = nth_error lb i) ->
    la = lb.

Lemma list_nth_some' :
  forall A (l:list A) i x,
    List.nth_error l i = Some x ->
    i < length l.

Lemma list_exist :
  forall A n (P:nat -> A -> Prop),
    ( forall i, i < n -> exists x, P i x ) ->
    exists l,
      length l = n /\
      (forall i x, nth_error l i = Some x -> P i x).

Lemma map_Equal_0 :
  forall T (m1 m2: M.t T),
    ( forall x, M.In x m1 <-> M.In x m2 ) ->
    ( forall x t1 t2,
        M.MapsTo x t1 m1 ->
        M.MapsTo x t2 m2 ->
        t1 = t2 ) ->
    M.Equal m1 m2.

Lemma map_exist :
  forall A B (mm: M.t B) (P: M.key -> A -> Prop) (Q: B -> Prop),
  ( forall b, { Q b } + { ~ Q b } ) ->
  ( forall x b, M.MapsTo x b mm -> Q b -> exists a, P x a ) ->
  exists m,
    ( forall x,
        (exists b, M.MapsTo x b mm /\ Q b) <->
        M.In x m ) /\
    ( forall x a,
        M.MapsTo x a m ->
        P x a ).
